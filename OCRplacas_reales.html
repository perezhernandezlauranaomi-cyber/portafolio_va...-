import cv2
import pytesseract

pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

"lista de placas aceptadas en el estacionamiento del CBtis144"
placas_autorizadas=["XZT-493","MNR-820","36HGG3","DRT-414-G","PCH-96-04"]
print("")
print("Placas registradas",*placas_autorizadas)

#cargar placa a leer
placa=cv2.imread("placa real1.png")
placanueva=cv2.resize(placa,(900,300))
placa2=cv2.imread("placa real2.png")
placanueva2=cv2.resize(placa2,(900,300))

#convertir a escala de grises
placa_gris=cv2.cvtColor(placanueva,cv2.COLOR_BGR2GRAY)
placa_gris2=cv2.cvtColor(placanueva2,cv2.COLOR_BGR2GRAY)


#separamos el fondo con el texto
placa_res=cv2.threshold(placa_gris,120,255,cv2.THRESH_BINARY)[1]
placa_res2=cv2.threshold(placa_gris2,120,255,cv2.THRESH_BINARY)[1]


#configurar OCR
placa_texto=pytesseract.image_to_string(placa_res,lang="eng")
placa_texto2=pytesseract.image_to_string(placa_res2,lang="eng")


#limpiamos la cadena de texto
placa_texto_limpio=placa_texto.strip().replace(" ","")
placa_texto_limpio2=placa_texto2.strip().replace(" ","")


#buscar la placa en la lista autorizada
if placa_texto_limpio in placas_autorizadas:
    print("ACCESO AUTORIZADO!! \nBIENVENIDO AL ESTACIONAMIENTO DEL CBTIS 144")
    print(placa_texto)
    if placa_texto_limpio in placas_autorizadas:
        placas_autorizadas.remove(placa_texto_limpio)
        print("Placas que no han ingresado: ",*placas_autorizadas)
    print("Placas que han ingresado: ",placa_texto_limpio)
    print("===================================================")

else:
    print("PLACA NO REGISTRADA 1")
    print(placa_texto)
    print(placas_autorizadas)

if placa_texto_limpio2 in placas_autorizadas:
    print("ACCESO AUTORIZADO!! \nBIENVENIDO AL ESTACIONAMIENTO DEL CBTIS 144")
    print(placa_texto2,"prueba")
    if placa_texto_limpio2 in placas_autorizadas:
        placas_autorizadas.remove(placa_texto_limpio2)
        print("Placas que no han ingresado: ",*placas_autorizadas)
        print("Placas que han ingresado: ", placa_texto_limpio,placa_texto_limpio2)

    print("===================================================")

else:
    print("PLACA SI REGISTRADA 2")
    print(placa_texto2)
    print("Placas que no han ingresado: ",placas_autorizadas)


cv2.imshow("imagen analizada",placanueva)
cv2.imshow("imagen analizada 2",placanueva2)
cv2.waitKey(0)
cv2.destroyAllWindows()
